DROP TABLE informix.ped_item_orig;

CREATE TABLE informix.ped_item_orig
	(
	cod_empresa   CHAR (2),
	num_pedido    DECIMAL(6,0),
	num_sequencia DECIMAL(5,0),
	cod_item      CHAR (15),
	pre_unit      DECIMAL(17,6)
	);

CREATE UNIQUE INDEX ix_pedito1
	ON ped_item_orig (cod_empresa, num_pedido, num_sequencia, cod_item);
